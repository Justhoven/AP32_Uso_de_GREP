1.  cat anillo.txt | grep agujero       (también vale grep agujero anillo.txt (para todos), muestra agujero, agujeroS, 2agujero... pero NO si tiene mayus)
2.  cat anillo.txt | grep -i agujero    (-i para ignorar mayus/minus)
3.  cat anillo.txt | grep -w anillo     (-w para vetar que tenga nada detrás o delante (salvo un punto), solo "anillo")
4.  cat anillo.txt | grep -nviw anillo  (-n para el número, -v para lineas en las que no aparezca)
5.  cat anillo.txt | grep ^Frodo        (^ = muestra líneas que empiecen por Frodo)
